#include<iostream>
#include<cstdio>
using namespace std;
int n,m,k,x,z,t,num,f[30005];
int getf(int x){
    if(f[x] == x)
        return f[x];
    return f[x] = getf(f[x]);
}
void merge(int x,int y){
    int tx = getf(x);
    int ty = getf(y);
    if(tx != ty)
        f[ty] = tx;
    return;
}
int main(){
    for(int i = 0;i < 30005;i++)
        f[i] = i;
    scanf("%d%d",&n,&m);
    for(int i = 0;i < m;i++){
        scanf("%d%d",&k,&x);
        if(i == 0)
            t = x;//标记读书社团中的一个人
        for(int j = 1;j < k;j++){
            scanf("%d",&z);
            merge(x,z);
        }
    }
    for(int i = 0;i < 30005;i++)
        if(getf(t) == getf(i))//找在一个集合里的
            num++;
    printf("%d\n",num);
    return 0;
}
