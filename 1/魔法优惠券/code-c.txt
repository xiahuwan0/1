#include <stdio.h>
#include <stdlib.h>

int compare(const void *a, const void *b) {
    return (*(int *)a - *(int *)b);
}

int main() {
    int n, m;
    scanf("%d", &n);
    int *coupon = (int *)malloc(n * sizeof(int));
    for (int i = 0; i < n; i++) {
        scanf("%d", &coupon[i]);
    }

    scanf("%d", &m);
    int *product = (int *)malloc(m * sizeof(int));
    for (int i = 0; i < m; i++) {
        scanf("%d", &product[i]);
    }

    qsort(coupon, n, sizeof(int), compare);
    qsort(product, m, sizeof(int), compare);

    int p = 0, q = 0, cost = 0;
    while (p < n && q < m && coupon[p] <= 0 && product[q] <= 0) {
        cost += coupon[p] * product[q];
        p++, q++;
    }

    p = n - 1;
    q = m - 1;
    while (p >= 0 && q >= 0 && coupon[p] > 0 && product[q] > 0) {
        cost += coupon[p] * product[q];
        p--, q--;
    }

    printf("%d\n", cost);

    free(coupon);
    free(product);

    return 0;
}
